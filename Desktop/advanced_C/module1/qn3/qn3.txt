#include <stdio.h>

int searchInMatrix(int *mat, int n, int key) {
    int row = 0;
    int col = n - 1;

    while (row < n && col >= 0) {
        int val = *((mat + row * n) + col); 
        if (val == key) {
            return 1; 
        } else if (val > key) {
            col--; 
        } else {
            row++; 
        }
    }
    return 0; 
}

int main() {
    int mat[4][4] = {
        {1,  4,  7, 11},
        {2,  5,  8, 12},
        {3,  6,  9, 16},
        {10,13,14,17}
    };
    int key;

    printf("Enter the key to search: ");
    scanf("%d", &key);

    if (searchInMatrix(&mat[0][0], 4, key)) {
        printf("Key %d found in matrix.\n", key);
    } else {
        printf("Key %d not found in matrix.\n", key);
    }

    return 0;
}
